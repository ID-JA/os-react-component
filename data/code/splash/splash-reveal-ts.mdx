name: TypeScript Reveal Splash
---

```tsx
function RevealSplash({ title }: { title: string }) {
  // Container width animation
  const containerWidth = {
    hidden: {
      width: "100%",
    },
    visible: {
      width: "0%",
      transition: { ease: [0.455, 0.03, 0.515, 0.955], duration: 1.5 },
    },
  };

  // Container children delay
  const delayChildren = {
    hidden: {
      opacity: 0,
    },
    visible: {
      opacity: 1,
      transition: {
        staggerChildren: 0.4,
      },
    },
  };

  // Fade In animation
  const fadeIn = {
    hidden: {
      opacity: 0,
      y: 100,
    },
    visible: {
      opacity: 1,
      y: 0,
      transition: { ease: [0.455, 0.03, 0.515, 0.955], duration: 2 },
    },
  };

  return (
    <motion.section
      initial='hidden'
      animate='visible'
      className='relative flex h-full items-center justify-center bg-gray-50 py-8 shadow-2xl dark:bg-gray-800'
    >
      <motion.div
        variants={delayChildren}
        className='absolute top-0 z-50 flex h-full w-full'
      >
        <motion.div
          variants={containerWidth}
          className='h-full bg-gradient-to-r from-blue-900 to-cyan-300'
        ></motion.div>
      </motion.div>
      <motion.h1
        variants={fadeIn}
        className='mb-6 max-w-3xl text-center font-cal text-xl dark:text-white lg:text-2xl lg:leading-tight'
      >
        {title}
      </motion.h1>
    </motion.section>
  );
}
```
